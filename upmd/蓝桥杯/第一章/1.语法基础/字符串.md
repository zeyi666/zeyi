```
	#include <iostream>
	#include <string>
	
	int main() {
		// 声明并初始化一个字符串
		std::string str1;
		
		// 使用字符串字面量初始化字符串
		std::string str2 = “Hello, World!”;
		
		// 使用另一个 std::string 对象来初始化字符串
		std::string str3 = str2;
		
		// 使用部分字符串初始化字符串
		std::string str4 = str2.substr(0,5);
		
		// 使用字符串数组初始化字符串
		const char* charArray = "Hello";
		std::string str5(charArray);
		
		// 使用重复的字符初始化字符串
		std::string str6(5,'A');
		
		// 输出字符串内容
		std::cout << "str1: " << str1 << std::endl;
		std::cout << "str2: " << str2 << std::endl;
		std::cout << "str3: " << str3 << std::endl;
		std::cout << "str4: " << str4 << std::endl;
		std::cout << "str5: " << str5 << std::endl;
		std::cout << "str6: " << str6 << std::endl;
		
		return 0;
	}
	

---
str.substr(0,5);
	第一个参数表示起始位置
	第二个参数表示长度 如果忽略参数2会自动获取到字符串末尾

string 是没有\0结尾的

string(str);
	可以直接复制字符串过来

string str(7,'C');
	将str初始化为 7 个相同的字符 'C'
```

基本操作
```
	在C++中,  std::string类提供了一个成员函数 c_str() , 用于返回一个指向以空字符结尾的C风格字符串 (即 const char *类型)
	在进行printf输出时,需要将string转换为C风格的字符串进行输出
```

```
#include <bits/stdc++.h>
using namespace std;

int main() {
	ios::sync_with_stdio(),cin.tie(0),cout.tie(0);
	
	const char str1[10] = "ABCDEFGHI";
	string str2(str1);
	
	cout << "str1:" << str1 << endl;
	cout << "str2:" + str2 + "\n";
	
	printf("Print str1:%s\n",str1);
	// printf("Print str2:%s\n",str2);
	// ↑这样会报错 因为str2是string类型 需要转换成char[]字符串数组才能输出
	printf("Print str2:%s\n",str2.c_str()); 
	
	return 0;
} 

---
s.c_str() : 将string类型的字符串转换为C语言的char[]字符串数组
```

1.获取字符串长度(length/size)
```
	std::string str = "Hello, World";
	int length = str.length();
	std::cout << "Length: " << length << std::endl;
```
2.拼接字符串 ( + 或 append )
```
std::string str1 = "Hello";
std::string str2 = "Hello!";
std::string result1 = str1 + ", " + str2; // 使用 + 运算符
std::string result2 = str1.append(", ").append(str2); // 使用 append() 函数
std::cout << "Result 1:" << result1 << std::endl;
std::cout << "Result 2:" << result2 << std::endl;

---
使用 + 并不会改变字符串内容
使用str.append() 会改变
	str.append() 会把字符串直接拼接进字符串里面存储 会改变字符串
```
3.字符串查找(find)
```
	std::string str = "Hello, World";
	size_t pos = str.find("World"); // 查找子字符串的位置
	if (pos != std::string::npos) {
		std::cout << "Substring found at position: " << pos << std::endl;
	} else {
		std::cout << "Substring not found." << std::endl;
	}
	
---
std::string::npos 是 -1 在这里用于判断出没有找到
```
4.字符串替换(replace)
```
	std::string str = "Hello, World!";
	str.replace(7, 5, "Universe"); // 替换子字符串
	std::cout << "Result: " << str << std::endl;
	

---
str.replace(7,5);
	第一个参数是起始位置
	第二个参数是要替换的长度
	第三个参数是要替换的文本
```

6.字符串比较 (compare)
```
	std::string str1 = "Hello";
	std::string str2 = "World";
	int result = str1.compare(str2); // 比较字符串
	if (result == 0) {
		std::cout << "Strings are equal," << std::endl;
	} else if (result < 0) {
		std::cout << "String 1 is less than String 2." << std::endl;
	} else {
		std::cout << "String 1 is greater than String 2." << std::endl;
	}
```

遍历string字符串内容
```
string s = "Hello";

// 方法1
for (int i = 0; i < s.length(); i++) {
	cout << s[i];
}

// 方法2
for (auto i : s) { // 执行的操作是将s复制出来然后复制给i
	cout << i;     // 所以这里直接输出i
	i = 'a';       // 此处修改无效 因为i是拷贝出来的 不是引用
}

// 方法3
for (auto &i : s) { // 执行的操作是直接将s引用给i
	cout << i;      // 这里相当于输出的s本身
	i = 'a';        // 此处修改会改变s的字符值 因为值引用的s
}
```

```
#include <bits/stdc++.h>
using namespace std;
 
int main() {
	ios::sync_with_stdio(),cin.tie(0),cout.tie(0);
	
	string str1 = "Hello";
	string str2 = "World";
	
	
	cout << "字符串长度: " << (str1 + str2).length() << '\n';
	
	cout << "拼接结果1: " << str1 + ", " + str2 + '\n';
	str1.append(", ").append(str2);
	cout << "拼接结果2: " << str1 << endl;
	
	char str_t1[] = " World";
	size_t start_ct = str1.find(str_t1);
	cout << "查找结果: "; 
	if (start_ct == string::npos) {
		cout << "未查询到" << endl;
	} 
	cout << start_ct << '\n'; 
	
	string str3 = "zeyi";
	str1.replace(7,5,str3);
	cout << "替换结果: " + str1 + "\n";
	
	string str4 = str1.substr(8,2);
	cout << "提取到的字符是:" << str4 << endl; 
	
	string str5 = "I LOVE YOU";
	string str6 = "i LOVE YOU";
	string str7 = "I LOVE YOU";
	
	size_t t1 = str6.compare(str5);
	size_t t2 = str6.compare(str7);

	} else {

	}

}
```
