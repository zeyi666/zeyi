#include <iostream>
using std::cout;
using std::cin;

struct node{
	int data;
	node* next;
};
node* head;

void insert(int a,int b) {
	 node* temp1 = (node*)malloc(sizeof(node*));
	 temp1->data = a;
	 temp1->next = NULL; 
	 // 插入到头部 
	 if (b == 1) {
	 	temp1->next = head;
	 	head = temp1;
	 	return; 
	 } 
	 // 插入到任意物位置 
	 node* temp2 = head;
	 for (int i = 0; i < b - 2; i++) {
	 	temp2 = temp2->next;
	 }
	 temp1->next = temp2->next;
	 temp2->next = temp1;
}

void Print() {
	node* temp = head;
	while (temp != NULL) {
		cout << temp->data << " ";
		temp = temp->next;
	}
	cout << '\n';
}

int main() {
	head = NULL; 
	int a,b;
	while (1) {
		cout << ':';
		cin >> a >> b;
		insert(a,b);
		Print();
	} 
}
