```
#include <iostream>
using std::cin;
using std::cout;

struct node{
	int data;
	node* next;
};
node* head,* tail;

void Insert(int age) {
	node* temp = new node();
	(*temp).data = age;
	(*temp).next = NULL; 
	if (head == NULL) head = temp;
	else tail->next = temp;
	tail = temp;
}

void Print_A(node *temp) {
	if (temp == NULL) return;
	cout << temp->data << " ";
	Print_A(temp->next);
}

void Print_B(node *temp) {
	if (temp == NULL) return;
	Print_B(temp->next);
	cout << temp->data << " ";
}

int main() {
	head = NULL,tail = NULL;
	int N;
	while (1) {
		cout << ":";
		cin >> N;
		if (N == 99) break; 
		Insert(N);
	}
	Print_A(head);
	cout << '\n';
	Print_B(head);
} 
```
***迭代方法会比递归方法高效很多
	- 迭代只会使用一个临时变量
	- 递归创建如此多的函数会占用内存的栈区空间
