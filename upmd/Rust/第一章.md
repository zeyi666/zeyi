		use std::io;
		fn main() {
			println!("猜数字游戏！");
			println!("请输入你的数字:");
			let mut a = String::new();
			io::stdin().read_line(&mut a).expect("无法读取行！");
			println!("您输入的数字是{}",a);
		}
		use
			-引用库 std库中的输入输出库io std::io
		fn 
			-定义一个函数 主函数main
		let mut a = String::new()
			-定义一个变量 变量名为a mut表示这个变量是可以被修改变动的 否则就是只读变量
				- let 定义变量 
				- mut 可以修改改变 
				- a 变量名 
				- = 赋值 
				- String标准库中的新创建 
				- ::访问子模块 ()表示关联函数
				- 关联函数名为 new() 表示创建空String对象
		io::stdin().read_line(&mut a).expect("无法读取行");
			- io模块中的 
			- stdin()输入函数 . 中的read_line(&mut a) 
			- read_line(&mut a)读取一行字符串并传递给变量 
			- .expect() 是Result类型的方法 用于处理的错误 成功返回结果 失败则panic并打印错误信息
		{}
			- 占位符  用于输出变量
