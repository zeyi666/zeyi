- 最后一个成员的指针是NULL（0）

链表的表示
![[链表2.png]]
```
struct node {
	int value;
	struct node *next;
}
这是一个链表的结点
``` 

```
#include <bits/stdc++.h>
using namespace std;

// 一个基础结点 
struct node {
	int age;
	struct node *next;
};

int main() {
	ios::sync_with_stdio(),cin.tie(0),cout.tie(0);
	
	// 创建结点对象a 和 b 
	struct node a;
	struct node b;
	
	// 为两个节点对象赋值 
	a.age = 17;
	b.age = 18;
	
	// 这是一个链表 它指向空 
	struct node *list_NULL = NULL;
	
	// 链表ab被串起来了
	struct node *list = &a;
	list->next = &b;
	list->next->next = NULL; 
	cout << "a age=" << list->age << " b age=" << list->next->age << '\n';
	
	struct node *list_b = &a;
	a.next = &b;
	a.next->next = NULL;
	cout << "a age=" << list_b->age << " b age=" << list_b->next->age << '\n';   
	
	return 0;
}
```
